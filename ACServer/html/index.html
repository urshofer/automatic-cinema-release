Automatic Cinema Server

This is the JSON API Server.

The Automatic Cinema Server stores the data, controls the narration and feeds controller, playback server and players.

	
/* 
	Login
	-----

	Creates a new Account if user and Name is not known.
	Returns a Session Id otherwise

 */

POST /Login

/* 
	Check Session
	-------------

	Checks if a session Id is Valid
	Returns true or an error

 */

GET /CheckSession/:checkSession

/* 
  	Store
	-----
  	Updates data
  
*/

POST /Store/:checkSession/:function

/* 
  	LoadOptions
	-----------
  	Loads Available Options
  
*/

GET /LoadOptions/:checkSession

/* 
  	LoadConfig
	----------
  	Sends the current configuration
  
*/

GET /LoadConfig/:checkSession

/* 
  	Channels
	--------
  	Loads Available Channels, without a channel list channels available
  
*/

GET /Channels/:checkSession/:channel
GET /Channels/:checkSession

/* 
  	Toggle
	------
  	Toggles the State of channels
  
*/


GET /Toggle/:checkSession/:channel/:state

/* 
  	Delete
	------
  	Deletes a show, style, target, channel
  
*/

DEL /Delete/:checkSession/:type/:name

/* 
  	Rename
	------
  	Renames a show, style, target, channel
  
*/

POST /Rename/:checkSession/:type/:name

/* 
  	Load
	----
    Loads a show, style, target, channel
  
*/

GET /Load/:checkSession/:type

/* 
  	StoreFile
	---------
    Stores a File with a multi part file form
  
*/

POST /StoreFile/:checkSession

/* 
  	Preview
	-------
    Sends a preview file
  
*/

GET /Preview/:checkSession/:thumb

/* 
  	Preview
	-------
    Sends a preview file
  
*/

GET /Download/:checkSession/:file

/* 
  	File List
	---------
    Sends a file list
  
*/

GET /List/:checkSession

/* 
  	Next
	----
  	Get Next Clip in Narration
  
*/

GET /Next/:checkSession/:channel

/* 
  	Reset
	-----
  	Resets the channel
  
*/

GET /Reset/:checkSession/:channel

/* 
  	Is Reset
	----
  	Get Next Clip in Narration
  
*/

POST /SyncState/:checkSession

/* 
  	Is New Timeline
	---------------
	Returns true if a timeline is restarted,
	otherwise false
  
*/

GET /IsNewTimeline/:checkSession/:channel

/* 
  	Timeline
	--------
	Returns the timeline data of a channel
  
*/

GET /Timeline/:checkSession/:channel

/* 
  	HasMaster
	---------------
	Returns false if a channel is master,
	otherwise returns the name of the master channel
  
*/

GET /HasMaster/:checkSession/:channel

/* 
  	Preroll
	-------
  	Sends the global preroll value in ms
  
*/

GET /Preroll/:checkSession

/* 
  	Rate
	----
  	Sends the global frames per second rate
  
*/

GET /Rate/:checkSession

/* 
  	Texttiming
	----------
  	Sends the global text timings: lines per screen and
  
*/

GET /Texttiming/:checkSession
